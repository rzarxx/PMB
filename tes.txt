    document.addEventListener('DOMContentLoaded', function() {
        const p1 = document.getElementById('pilihan1');
        const p2 = document.getElementById('pilihan2');
        const errorMsg = document.getElementById('error-message');
        const submitBtn = document.getElementById('submit-btn');

        function validateSelection() {
            if (p1.value && p2.value && p1.value === p2.value) {
                errorMsg.classList.remove('hidden');
                
                p1.classList.add('border-error', 'focus:ring-error', 'focus:border-error');
                p1.classList.remove('border-outline-variant', 'focus:ring-primary', 'focus:border-primary');
                
                p2.classList.add('border-error', 'focus:ring-error', 'focus:border-error');
                p2.classList.remove('border-outline-variant', 'focus:ring-primary', 'focus:border-primary');
                
                submitBtn.disabled = true;
                submitBtn.classList.add('opacity-50', 'cursor-not-allowed');
            } else {
                errorMsg.classList.add('hidden');
                
                p1.classList.remove('border-error', 'focus:ring-error', 'focus:border-error');
                p1.classList.add('border-outline-variant', 'focus:ring-primary', 'focus:border-primary');
                
                p2.classList.remove('border-error', 'focus:ring-error', 'focus:border-error');
                p2.classList.add('border-outline-variant', 'focus:ring-primary', 'focus:border-primary');
                
                submitBtn.disabled = false;
                submitBtn.classList.remove('opacity-50', 'cursor-not-allowed');
            }
        }

        p1.addEventListener('change', validateSelection);
        p2.addEventListener('change', validateSelection);
    });
